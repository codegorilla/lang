
declaration:
  valueDecl
  variableDecl
  functionDecl
  classDecl

functionDecl:
  'def' identifier '(' parameters? ')' = (blockExpr | expression ';')

parameters:
  parameter (',' parameter)*

parameter:
  identifier

statement:
  expression? ';'

expression:
  assignmentExpr
  printExpr
  returnExpr
  whileExpr

whileExpr:
  'while' '(' expression ')' (blockExpr | expression ';')


assignmentExpr:
  blockExpr
  ifExpr


blockExpr:
  '{' blockElement* '}'

arrayLiteral:
  '[' arrayElements? ']'

arrayElements:
  expression (',' expression)*
